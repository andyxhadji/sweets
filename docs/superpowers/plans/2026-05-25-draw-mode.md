def test_render_truncates_oversized_grid():
    """Grid larger than board is truncated."""
    grid = [[1] * 20 for _ in range(10)]  # 10x20, larger than 3x15

    board = mode.render()
    arr = board.to_array()

"""Drawing display mode for Vestaboard."""

from typing import Any

from sweets.core.board import Board
from sweets.modes.base import Mode


class DrawingMode(Mode):
    """Display a user-drawn grid on the board."""

    name = "Drawing"
    slug = "drawing"
    interval = None  # Static, no periodic updates

    def __init__(self, rows: int = 6, cols: int = 22) -> None:
        super().__init__(rows, cols)
        self.grid: list[list[int]] = []

    def configure(self, settings: dict[str, Any]) -> None:
        """Apply mode-specific settings from config."""
        self.grid = settings.get("grid", [])

    def render(self) -> Board:
        """Render the stored grid to the board."""
        board = Board(rows=self.rows, cols=self.cols)
        for i, row in enumerate(self.grid[: self.rows]):
            board.set_row(i, row[: self.cols])
        return board
```

let grid = [];

function codeToDisplay(code) {
    if (code === 0) return {text: '', className: ''};
    if (code === 62) return {text: '♥', className: 'heart'};
    if (code >= 63 && code <= 70) return {text: '', className: 'color-' + code};
    if (code >= 1 && code <= 26) return {text: String.fromCharCode(64 + code), className: 'filled'};
    if (code >= 27 && code <= 35) return {text: String(code - 26), className: 'filled'};
    if (code === 36) return {text: '0', className: 'filled'};
    return {text: '?', className: 'filled'};
}

function codeToName(code) {
    if (code === 0) return 'Eraser';
    if (code === 62) return 'Heart';
    if (code === 63) return 'Red';
    if (code === 64) return 'Orange';

    if (code >= 27 && code <= 35) return String(code - 26);
    if (code === 36) return '0';
    return '?';
}

// Update cell display
function updateCell(row, col) {
    const cell = document.querySelector(`[data-row="${row}"][data-col="${col}"]`);
    if (!cell) return;

    const code = grid[row][col];
    const display = codeToDisplay(code);

    // Reset classes
    cell.className = 'cell';
    if (display.className) {
        cell.classList.add(display.className);
    }
    cell.textContent = display.text;
}

// Paint a cell
function paintCell(row, col) {
    grid[row][col] = selectedBrush;
    updateCell(row, col);
}

// Get cell from touch/mouse event
function getCellFromPoint(x, y) {
    const el = document.elementFromPoint(x, y);
    if (el && el.classList.contains('cell') && el.dataset.row !== undefined) {
        return {row: parseInt(el.dataset.row), col: parseInt(el.dataset.col)};
    }
    return null;
}

@app.route("/draw/send", methods=["POST"])
def draw_send():
    """Send a drawing to the board."""
    sched = get_scheduler()
    data = request.get_json()

    if not data or "grid" not in data:
        return jsonify({"error": "Missing grid data"}), 400

    grid = data["grid"]

    # Validate grid is a list of lists of ints
    if not isinstance(grid, list):
        return jsonify({"error": "Grid must be a list"}), 400

    for row in grid:
        if not isinstance(row, list):
            return jsonify({"error": "Grid rows must be lists"}), 400
        for val in row:
            if not isinstance(val, int) or val < 0 or val > 70:
                return jsonify({"error": "Grid values must be integers 0-70"}), 400

    # Store grid in mode settings and start drawing mode
    sched.mode_settings["drawing"] = {"grid": grid}
    try:
        sched.start("drawing")
    except RateLimitError:
        return jsonify({"error": "Rate limited - please wait ~15 seconds"}), 429
    except QuietHoursError:
        return jsonify({"error": "Vestaboard is in quiet hours"}), 423

    return jsonify({"success": True})
```
